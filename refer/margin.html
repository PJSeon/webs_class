CSS margin

margin 설정 방법

margin-top: 10px;

margin-right: 11px;

margin-bottom: 12px;

margin-left: 13px;              //*시계방향으로 값이 주어진다 *//
  
margin: 10px;                   //* 위, 오른쪽, 아래, 왼쪽 : 10px *//

margin: 10px 11px;              //* 위, 아래:10px, 왼쪽, 오른쪽 11px *//

margin: 10px 11px 12px;         //* 위:10px, 왼쪽,오른쪽:11px, 아래:12px *//

margin: 10px 11px 12px 13px;    //* 위:10px, 오른쪽:11px, 아래:12px, 왼쪽:13px *//



margin: auto;                   //* 자동값, 기본값, padding에는 없음*//

margin: 0 auto;                 //* 가운데 정렬(블록구조 width값이 있을 때에만) *//
