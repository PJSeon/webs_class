01. 변수 : 데이터 저장

 {
    var x = 100;                 //변수 x에 숫자 100을 저장
    var y = 200;                //변수 y에 숫자 200을 저장
    var z = "javascript";       //변수 z에 문자 javascript 저장, 문자 입력시에는 " "

    document.write("<br>*** 01. 변수 ***<br>");
    document.write(x, "<br>");
    document.write(y, "<br>");
    document.write(z, "<br>");
}

02. 변수 : 데이터 저장 + 데이터 변경

 {
    let x = 100;
    let y = 200;
    let z = "javascript";
    x = 300;
    y = 400;
    z = "jquery"
    
    document.write("*** 02. 변수 ***<br>");
    document.write(x, "<br>");
    document.write(y, "<br>");
    document.write(z, "<br>");
}

03. 변수 : 데이터 저장 + 데이터 변경 + 데이터 추가

 {
    let x = 100;
    let y = 200;
    let z = "javascript";
    
    x += 300;
    y += 400;
    z += "jquery";

    document.write ("<br>*** 03. 변수 *** <br>");
    document.write (x, "<br>");
    document.write (y, "<br>");
    document.write (z, "<br>");
}
