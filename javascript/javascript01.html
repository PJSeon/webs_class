01. 변수 : 데이터 저장

{
    var x = 100;                 //변수 x에 숫자 100을 저장
    var y = 200;                //변수 y에 숫자 200을 저장
    var z = "javascript";       //변수 z에 문자 javascript 저장, 문자 입력시에는 " "

    document.write("<br>*** 01. 변수 ***<br>");
    document.write(x, "<br>");
    document.write(y, "<br>");
    document.write(z, "<br>");
}

200

03. 변수 : 데이터 저장 + 데이터 변경 + 데이터 추가

{

    let y = 200;
    let z = "javascript";

    document.write ("<br>*** 03. 변수 *** <br>");

    document.write (z, "<br>");
}

04. 변수 : 변수의 종류 : 지역변수 + 전역변수 + 매개변수

 {
    //변수 : 데이터 저장
    //함수 : 데이터 실행

    document.write("
*** 04. 변수 : 변수의 종류 ***
");

    let x = 100;    //전역변수
    let y = 200;    //전역변수

    function func() {

        let x = 100;    //지역변수
        let z = "javascript";
        x = 200;    //지역변수 100 -> 200
        y = 300;    //전역변수 200 -> 300

        document.write("<br>함수 안<br>");
        document.write(x, "<br>");
        document.write(y, "<br>");
        document.write(z, "<br>");
    }
    func();

    document.write("함수 밖 <br>")
    document.write(x, "<br>");
    document.write(y, "<br>");
    document.write(z, "<br>");
}

결과보기

함수 안
200
300
javascript
함수 밖
100
300
